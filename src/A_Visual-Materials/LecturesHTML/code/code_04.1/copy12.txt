public class EList {
	static class ListItem {
		
		private ListItem next;
		
		void setNext(ListItem item) {
			next = item;
		}
		public ListItem getNext() {
			return  next;
		}
	}
	private ListItem head;
	private ListItem tail;
	
	public ListItem getHead() {
		return  head;
	}
	public void add(ListItem item) {
		if(head == null) {
			head = item;
			tail = item;
		} else {
			tail.setNext(item);
			tail = item;
		}
	}
}
