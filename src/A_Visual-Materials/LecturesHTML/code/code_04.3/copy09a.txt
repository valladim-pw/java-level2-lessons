		BinaryHeapMax2<Integer> heapI = new BinaryHeapMax2<>();
		
		for (int i = 0; i < 13; i++) {
			heapI.add(i);			
		}

		String str = "БУМАГОПРЯДИЛЬНЫЙ";

		}
		heapS.printHeap(3);
	}
