import java.util.Arrays;

public class SafeGeneric<T> {
	private T[] items;
	private int count;
	
	public SafeGeneric(){
		items = (T[])new Object[10];
		count = 0;
	}
	
	static class A {
		public A() {}
		
		@Override
		public String toString() {
			return 	"class A";
		}
	}
	
	public void add(T item) {		
		items[count++] = item;
	}
	
	public T get(int idx) {		
		return items[idx];
	}
	
	@Override
	public String toString() {
		return 	"items=" + Arrays.toString(items);
	}
	
	public static void main(String[] args) {
		SafeGeneric<String> nsg = new SafeGeneric();
		
		String first = "Hello";
		nsg.add(first);
		Integer val = 10;
		nsg.add(val);		
		
		System.out.println(nsg.toString());
	}
}
